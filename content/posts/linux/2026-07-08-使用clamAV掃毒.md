使用 clamAV 掃描家目錄

```bash
# 建立隔離區目錄
mkdir ~/quarantine

# 掃描家目錄 參數如下
# -r 遞迴
# -i 發現問題檔案時顯示資訊
# --bell 發現問題檔案時發出警示聲
# --move 將有問題的檔案移到隔離區
# -l 生成掃描 log 檔
# --exclude-dir 略過目錄(避免過大的資料)
clamscan -ri ~/ --bell --move=$HOME/quarantine -l $HOME/clamav_scan.log \
--exclude-dir="^\.cache" \
--exclude-dir="^\.local" \
--exclude-dir="Backups" \
--exclude-dir="quarantine"
```
